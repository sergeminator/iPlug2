61 inline agg::comp_op_e AGGBlendMode(const IBlend* pBlend)
62 {
63  if (!pBlend)
64  return agg::comp_op_src_over;
65 
66  switch (pBlend->mMethod)
67  {
68  case kBlendDefault: // fall through
69  case kBlendClobber: // fall through
70  case kBlendSourceOver: return agg::comp_op_src_over;
71  case kBlendSourceIn: return agg::comp_op_src_in;
72  case kBlendSourceOut: return agg::comp_op_src_out;
73  case kBlendSourceAtop: return agg::comp_op_src_atop;
74  case kBlendDestOver: return agg::comp_op_dst_over;
75  case kBlendDestIn: return agg::comp_op_dst_in;
76  case kBlendDestOut: return agg::comp_op_dst_out;
77  case kBlendDestAtop: return agg::comp_op_dst_atop;
78  case kBlendAdd: return agg::comp_op_plus;
79  case kBlendXOR: return agg::comp_op_xor;

129 void IGraphicsAGG::Rasterizer::Rasterize(const IPattern& pattern, agg::comp_op_e op, float opacity, EFillRule rule)
130 {
131  mRasterizer.filling_rule(rule == kFillWinding ? agg::fill_non_zero : agg::fill_even_odd );
132 
133  switch (pattern.mType)
134  {
135  case kSolidPattern:
136  {
137  // Rasterize
138 
139  Rasterize(AGGColor(pattern.GetStop(0).mColor, opacity), op);
140  }
141  break;
142 
143  case kLinearPattern:
144  case kRadialPattern:
145  {
146  // Common gradient objects
147 
148  const IMatrix& m = pattern.mTransform;
149 
150  agg::trans_affine gradientMTX(m.mXX, m.mYX , m.mXY, m.mYY, m.mTX, m.mTY);
151  agg::gradient_lut<agg::color_interpolator<agg::rgba8>, 512> colorArray;
152 
153  // Scaling
154 
155  gradientMTX = (agg::trans_affine() / mGraphics.mTransform) * gradientMTX * agg::trans_affine_scaling(512.0);
156 
157  // Make gradient lut
158 
159  colorArray.remove_all();
160 
161  for (int i = 0; i < pattern.NStops(); i++)
162  {
163  const IColorStop& stop = pattern.GetStop(i);
164  float offset = stop.mOffset;
165  colorArray.add_color(offset, AGGColor(stop.mColor, opacity));
166  }
167 
168  colorArray.build_lut();
169 
170  // Rasterize
171 
172  if (pattern.mType == kLinearPattern)
173  {
174  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_y(), gradientMTX, colorArray, op);
175  }
176  else
177  {
178  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_radial_d(), gradientMTX, colorArray, op);
179  }
180  }
181  break;
182  }
183 }

246 bool CheckTransform(const agg::trans_affine& mtx)
247 {
248  if (!agg::is_equal_eps(mtx.tx - std::round(mtx.tx), 0.0, 1e-3))

259 void IGraphicsAGG::DrawBitmap(const IBitmap& bitmap, const IRECT& dest, int srcX, int srcY, const IBlend* pBlend)
260 {
261  bool preMultiplied = static_cast<AGGBitmap*>(bitmap.GetAPIBitmap())->IsPreMultiplied();
262  IRECT bounds = mClipRECT.Empty() ? dest : mClipRECT.Intersect(dest);
263  bounds.Scale(GetBackingPixelScale());
264 
265  APIBitmap* pAPIBitmap = dynamic_cast<AGGBitmap*>(bitmap.GetAPIBitmap());
266  agg::pixel_map* pSource = pAPIBitmap->GetBitmap();
267  agg::rendering_buffer src(pSource->buf(), pSource->width(), pSource->height(), pSource->row_bytes());
268 
269  agg::trans_affine srcMtx;
270  srcMtx /= mTransform;
271  srcMtx *= agg::trans_affine_translation(srcX - dest.L, srcY - dest.T);
272  srcMtx *= agg::trans_affine_scaling(bitmap.GetScale() * bitmap.GetDrawScale());
273 
274  if (bounds.IsPixelAligned() && CheckTransform(srcMtx))
275  {
276  double scale = GetScreenScale() * GetScreenScale() / pAPIBitmap->GetScale();
277  IRECT destScaled = dest.GetScaled(GetBackingPixelScale());
278  srcX = std::round(srcX * scale + std::max(0.f, bounds.L - destScaled.L));
279  srcY = std::round(srcY * scale + std::max(0.f, bounds.T - destScaled.T));
280  bounds.Translate(mTransform.tx, mTransform.ty);
281 
282  mRasterizer.BlendFrom(src, bounds, srcX, srcY, AGGBlendMode(pBlend), AGGCover(pBlend), preMultiplied);
283  }
284  else
285  {
286  agg::rounded_rect rect(dest.L, dest.T, dest.R, dest.B, 0);
287  agg::conv_transform<agg::rounded_rect> tr(rect, mTransform);
288 
289  if (preMultiplied)
290  {
291  PixfmtPreType fmtSrc(src);
292  mRasterizer.Rasterize(fmtSrc, tr, srcMtx, AGGBlendMode(pBlend), AGGCover(pBlend));
293  }
294  else
295  {
296  PixfmtType fmtSrc(src);
297  mRasterizer.Rasterize(fmtSrc, tr, srcMtx, AGGBlendMode(pBlend), AGGCover(pBlend));
298  }
299  }

373 void IGraphicsAGG::PathStroke(const IPattern& pattern, float thickness, const IStrokeOptions& options, const IBlend* pBlend)
374 {
375  typedef agg::conv_curve<agg::path_storage> CPType;
376  typedef agg::conv_transform<CPType> S1Type;
377  typedef agg::conv_stroke<S1Type> S2Type;
378  typedef agg::conv_transform<S2Type> S3Type;
379  typedef agg::conv_dash<S1Type> D2Type;
380  typedef agg::conv_stroke<D2Type> D3Type;
381  typedef agg::conv_transform<D3Type> D4Type;
382 
383  agg::trans_affine tranform(mTransform);
384  CPType curvedPath(mPath);
385  S1Type basePath(curvedPath, tranform.invert());
386 
387  if (options.mDash.GetCount())
388  {
389  D2Type dashedPath(basePath);
390  D3Type strokedDashedPath(dashedPath);
391  D4Type finalPath(strokedDashedPath, mTransform);
392 
393  // Set the dashes (N.B. - for odd counts the array is read twice)
394 
395  int dashCount = options.mDash.GetCount();
396  int dashMax = dashCount & 1 ? dashCount * 2 : dashCount;
397  const float* dashArray = options.mDash.GetArray();
398 
399  dashedPath.remove_all_dashes();
400  dashedPath.dash_start(options.mDash.GetOffset());
401 
402  for (int i = 0; i < dashMax; i += 2)
403  dashedPath.add_dash(dashArray[i % dashCount], dashArray[(i + 1) % dashCount]);
404 
405  StrokeOptions(strokedDashedPath, thickness, options);
406  mRasterizer.Rasterize(finalPath, pattern, AGGBlendMode(pBlend), BlendWeight(pBlend));
407  }
408  else
409  {
410  S2Type strokedPath(basePath);
411  S3Type finalPath(strokedPath, mTransform);
412 
413  StrokeOptions(strokedPath, thickness, options);
414  mRasterizer.Rasterize(finalPath, pattern, AGGBlendMode(pBlend), BlendWeight(pBlend));
415  }
416 
417  if (!options.mPreserve)
418  mPath.remove_all();
419 }

585 bool IGraphicsAGG::DoDrawMeasureText(const IText& text, const char* str, IRECT& bounds, const IBlend* pBlend, bool measure)
586 {
587  if (!str || str[0] == '\0')
588  {
589  return false;
590  }
591 
592  float weight = 0.0;
593  const bool kerning = false;
594  const bool hinting = false;
595 
596  mFontContour.width(-weight * (text.mSize * 0.05));
597 
598  StaticStorage<IFontData>::Accessor storage(sFontCache);
599  IFontData* pFont = storage.Find(text.mFont);
600 
601  if (!pFont || !SetFont(text.mFont, pFont))
602  {
603  assert(0 && "No font found - did you forget to load it?");
604  }
605 
606  mFontEngine.hinting(hinting);
607  mFontEngine.height(text.mSize);
608  mFontEngine.width(text.mSize);
609  mFontEngine.flip_y(true);
610 
611  WDL_TypedBuf<LineInfo> lines;
612  CalculateTextLines(&lines, bounds, str, mFontManager);
613  LineInfo * pLines = lines.Get();
614 
615  double x = bounds.L;
616  double y = bounds.T + (text.mSize);
617 
618  switch (text.mVAlign)
619  {
620  case IText::kVAlignTop: y = bounds.T + mFontEngine.ascender(); break;
621  case IText::kVAlignMiddle: y = bounds.MH() + mFontEngine.descender() + text.mSize/2.; break;
622  case IText::kVAlignBottom: y = bounds.B + mFontEngine.descender(); break;
623  }
624 
625  if (measure)
626  {
627  double width = 0.0;
628 
629  for (int i = 0; i < lines.GetSize(); ++i, ++pLines)
630  width = std::max(width, pLines->mWidth);
631 
632  bounds.B = bounds.T + lines.GetSize() * text.mSize;
633  bounds.R = bounds.L + width;
